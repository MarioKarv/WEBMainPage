Завдання №7:

Напиши клас Client який створює об'єкт з властивостями login email Оголоси приватні властивості #login #email, доступ до яких зроби через геттер та сеттер login email

Код JavaScript:

class Client { #login; #email; constructor(login, email) { this.#login = login; this.#email = email; } get login() { return this.#login; } set login(newLogin) { this.#login = newLogin; } get email() { return this.#email; } set email(newEmail) { this.#email = newEmail; } } // Створення нового об'єкта класу Client let client = new Client("userLogin", "userEmail@example.com"); // Виведення властивостей об'єкта в консоль console.log(client.login); // "userLogin" console.log(client.email); // "userEmail@example.com" // Зміна властивостей об'єкта client.login = "newUserLogin"; client.email = "newUserEmail@example.com"; // Виведення змінених властивостей об'єкта в консоль console.log(client.login); // "newUserLogin" console.log(client.email); // "newUserEmail@example.com"
